## 한글 배너를 만들어서 위에서 아래쪽으로 왔다 갔다 하며 키보드로 속도 조절이 가능

```

PFont f; //한글 배너 만들기
void setup() {
  size(1200,400); // 화면 가로 세로 크기
  f = createFont("굴림", 80); //굴림체 80포인트
  textFont(f); // 한글 배너를 받는다
}
int i, dir=1, sp=1; // 정수 변수의 선언
void draw() {
  fill(0,0,255); // 파란색 글씨
  background(174,221,60); //녹색 배경
  text("그래픽스 재미있어요!", 200, i); //한글 배너
  if (i>height) dir=-1; // i의 값이 높이보다 커지면 다시 돌아간다
  if (i<60) dir=1; // i가 60보다 작아지면 다시 돌아간다
  i = i+dir*sp; // i = i의 값 + 키보드 숫자값으로 움직인다
}
void keyPressed() {
  sp = key-'0'; // 키보드 0~9까지 숫자를 눌러 속도 조절
}

```
